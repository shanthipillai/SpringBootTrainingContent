package feb.training.springboot.controller; import com.fasterxml.jackson.databind.JsonNode; import com.fasterxml.jackson.databind.ObjectMapper; import org.springframework.beans.factory.annotation.Value; import org.springframework.stereotype.Controller; import org.springframework.ui.Model; import org.springframework.web.bind.annotation.GetMapping; import org.springframework.web.bind.annotation.RequestParam; import org.springframework.web.client.RestTemplate; import java.util.ArrayList; import java.util.List; @Controller public class MarsImagePageController { @Value("${nasa.api.key}") private String nasaApiKey; private final String NASA_API_URL = "https://api.nasa.gov/mars-photos/api/v1/rovers/curiosity/photos"; @GetMapping("/mars-gallery") public String getMarsImages(@RequestParam(defaultValue = "1000") int sol, Model model) { RestTemplate restTemplate = new RestTemplate(); String url = NASA_API_URL + "?sol=" + sol + "&api_key=" + nasaApiKey; try { String response = restTemplate.getForObject(url, String.class); ObjectMapper mapper = new ObjectMapper(); JsonNode root = mapper.readTree(response); List imageUrls = new ArrayList<>(); for (JsonNode photo : root.get("photos")) { imageUrls.add(photo.get("img_src").asText()); } model.addAttribute("images", imageUrls); return "mars-gallery"; // maps to a Thymeleaf template } catch (Exception e) { model.addAttribute("error", "Error fetching NASA images: " + e.getMessage()); return "error"; } } }
