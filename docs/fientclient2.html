Spring Boot Feign Client - Advanced Complete Notes

Basic Feign Annotation

@FeignClient(name = "card-service", url = "http://localhost:8081")

Core Attributes

name (MANDATORY)

name = "card-service"

url (when NO Eureka)

url = "http://localhost:8081"

fallback

fallback = CardClientUnavailable.class

fallbackFactory (ADVANCED)

fallbackFactory = CardFallbackFactory.class

Configuration Properties

Timeout

feign.client.config.default.connectTimeout=2000
feign.client.config.default.readTimeout=5000

Logging

logging.level.com.bank.account.feign=DEBUG
logging.level.feign=DEBUG

Retry

spring.cloud.openfeign.client.config.default.retryer=feign.Retryer$Default

Disable Retry

spring.cloud.openfeign.client.config.default.retryer=feign.Retryer$NeverRetry

Circuit Breaker

spring.cloud.openfeign.circuitbreaker.enabled=true

Advanced Attributes

path

@FeignClient(name="card-service", url="http://localhost:8081", path="/cards")

configuration

@FeignClient(name="card-service", configuration = FeignConfig.class)

qualifiers

Used when multiple beans exist

Headers

Static Header

@GetMapping(value = "/account/{accountId}", headers = "Content-Type=application/json")

Dynamic Header

@GetMapping("/account/{accountId}")
List<Card> getCards(@RequestHeader("Authorization") String token,
                    @PathVariable int accountId);

Request Interceptor

@Bean
public RequestInterceptor requestInterceptor() {
    return requestTemplate -> {
        requestTemplate.header("Authorization", "Bearer token");
    };
}

Error Decoder

@Bean
public ErrorDecoder errorDecoder() {
    return new CustomErrorDecoder();
}

Clean Best Practice Example

@FeignClient(
    name = "card-service",
    url = "http://localhost:8081",
    path = "/cards",
    fallback = CardClientUnavailable.class
)
public interface CardClient {

    @GetMapping("/account/{accountId}")
    List<Card> getCards(@PathVariable("accountId") int accountId);
}
